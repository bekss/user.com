# Installation guide:  
  python -m venv env  
  source env/Scripts/activate  
  pip install -r requirements.txt  
## After installation all library:  
	pyhton manage.py migrate
  python manage.py createsuperuser
  python manage.py makemigrations
  python manage.py migrate
